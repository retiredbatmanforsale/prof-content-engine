import numpy as np

def _forward(x, W, b):
    return x @ W + b


def _loss(x, W, b):
    # Sum-of-elements loss makes d_loss/d_y = ones — handy for grad checks.
    return _forward(x, W, b).sum()


def _numerical_grad_w(x, W, b, eps=1e-5):
    g = np.zeros_like(W)
    it = np.nditer(W, flags=['multi_index'], op_flags=['readwrite'])
    while not it.finished:
        i, j = it.multi_index
        W[i, j] += eps; lp = _loss(x, W, b)
        W[i, j] -= 2 * eps; lm = _loss(x, W, b)
        W[i, j] += eps  # restore
        g[i, j] = (lp - lm) / (2 * eps)
        it.iternext()
    return g


@__test__("Returns three arrays with correct shapes")
def t():
    rng = np.random.default_rng(0)
    B, I, O = 4, 3, 5
    x = rng.standard_normal((B, I))
    W = rng.standard_normal((I, O))
    grad_out = rng.standard_normal((B, O))
    out = linear_backward(grad_out, x, W)
    assert out is not None and len(out) == 3, "expected (dx, dW, db) tuple"
    dx, dW, db = out
    assert dx.shape == (B, I), f"dx shape {dx.shape} != ({B}, {I})"
    assert dW.shape == (I, O), f"dW shape {dW.shape} != ({I}, {O})"
    assert db.shape == (O,), f"db shape {db.shape} != ({O},)"

import numpy as np

def linear_backward(grad_out, x, W):
    dx = grad_out @ W.T          # (B, out) @ (out, in) -> (B, in)
    dW = x.T @ grad_out          # (in, B) @ (B, out)   -> (in, out)
    db = grad_out.sum(axis=0)    # sum across the batch dimension
    return dx, dW, db
`}
  hints={[

    "Why those shapes? Forward is y = x @ W + b. By the chain rule: dx = (d y / d x) @ grad_out, and the Jacobian d y / d x is W (transposed for the matmul to align). Same logic for dW. The bias adds to every batch row, so its gradient sums across the batch.",
    "Sanity check by shape alone: grad_out is (B, out), W is (in, out). Only `grad_out @ W.T` produces (B, in). The shapes force the right answer.",
  ]}
>

Backward through Linear is the most-asked backprop derivation in interviews — and the one most candidates fumble. The math falls out of the chain rule on the matmul. Three gradients, three lines.

Forward (problem 01): `y = x @ W + b` where `x: (B, in)`, `W: (in, out)`, `b: (out,)`.

Backward (this problem): given `grad_out = ∂L / ∂y` of shape `(B, out)`, compute:

`dx = grad_out @ W.T              # (B, in)`

`dW = x.T @ grad_out              # (in, out)`

`db = grad_out.sum(axis=0)         # (out,) — sum across batch`

The harness checks each gradient independently and finishes with a finite-difference cross-check on `dW` (the trickiest of the three). Get the shapes right and the algebra is forced — there's exactly one way to multiply these matrices that matches the dimensions.
