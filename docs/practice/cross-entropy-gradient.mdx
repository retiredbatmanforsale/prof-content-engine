import numpy as np

def _softmax(x):
    s = x - x.max()
    e = np.exp(s)
    return e / e.sum()

def _ce_loss(logits, target):
    return -np.log(_softmax(logits)[target] + 1e-30)

def _numerical_grad(logits, target, eps=1e-4):
    g = np.zeros_like(logits, dtype=float)
    for i in range(len(logits)):
        h = logits.copy().astype(float)
        h[i] += eps; lp = _ce_loss(h, target)
        h[i] -= 2 * eps; lm = _ce_loss(h, target)
        g[i] = (lp - lm) / (2 * eps)
    return g

@__test__("Output shape matches logits")
def t():
    logits = np.array([1.0, 2.0, 3.0, 0.5])
    g = cross_entropy_grad(logits, target=2)
    assert g is not None, "function returned None"
    assert g.shape == logits.shape, f"shape {g.shape} vs {logits.shape}"

@__test__("Matches the analytic formula softmax - one_hot")
def t():
    logits = np.array([1.0, 2.0, 3.0, 0.5])
    g = cross_entropy_grad(logits, target=2)
    expected = _softmax(logits)
    expected[2] -= 1.0
    assert np.allclose(g, expected, atol=1e-7), f"got {g}, expected {expected}"

@__test__("Matches finite-difference numerical gradient")
def t():

    logits = np.array([0.5, -1.0, 2.5, 1.0, -0.3])
    for target in [0, 2, 4]:

import numpy as np

def cross_entropy_grad(logits, target):
    # Stable softmax.
    shifted = logits - np.max(logits)
    exps = np.exp(shifted)
    probs = exps / exps.sum()

    # dL/d(logits) = softmax(logits) - one_hot(target)
    grad = probs.copy()
    grad[target] -= 1.0
    return grad
`}

Cross-entropy is the loss every classifier and language model is trained against. The gradient of cross-entropy with respect to logits is one of those rare derivations where heavy calculus collapses into a one-liner — and that one-liner shows up everywhere from logistic regression to the final layer of GPT-4.

The setup. For a single example with logits $z \\in \\mathbb{R}^V$ and target class $y$:

$$
L = -\\log p_y, \\quad \\text{where } p = \\text{softmax}(z)
$$

The result. Take the derivative w.r.t. each logit $z_i$:

$$
\\frac{\\partial L}{\\partial z_i} = p_i - \\mathbf{1}[i = y]
$$

That is: **softmax(z) minus the one-hot encoding of y**. Implement that.

Two things the harness checks beyond correctness:

- **Numerical stability.** Same trick as the softmax problem — subtract the max before exp, or your gradient will NaN on real LLM-scale logits.
- **Agreement with a numerical gradient.** A finite-difference reference verifies you derived the analytic gradient correctly, not just memorized the formula. This is exactly how you'd debug a backprop bug in real life.
